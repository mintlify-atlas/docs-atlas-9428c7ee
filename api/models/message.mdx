## Overview

The `Message` class represents a message sent in a Fluxer channel. It provides methods for replying, editing, deleting, and managing reactions.

## Class Definition

```python
from fluxer.models import Message

@dataclass(slots=True)
class Message:
    """Represents a message in a Fluxer channel."""
```

## Usage Example

```python
import fluxer
from fluxer import Embed

client = fluxer.Client(token="your_token")

@client.event
async def on_message(message):
    # Ignore messages from bots
    if message.author.bot:
        return
    
    if message.content == "!info":
        # Create an embed with message info
        embed = Embed(
            title="Message Info",
            description=f"Message ID: {message.id}",
            color=0x5865F2
        )
        embed.add_field(name="Author", value=message.author.mention)
        embed.add_field(name="Channel", value=f"<#{message.channel_id}>")
        embed.add_field(name="Created", value=str(message.created_at))
        
        if message.edited_timestamp:
            embed.add_field(name="Edited", value=message.edited_timestamp)
        
        await message.reply(embed=embed)
    
    elif message.content == "!react":
        # Add multiple reactions
        await message.add_reaction("👍")
        await message.add_reaction("❤️")
        await message.add_reaction("🎉")
    
    elif message.content.startswith("!edit"):
        # Send a message and edit it
        msg = await message.channel.send("Original message")
        await msg.edit("Edited message!")
    
    elif message.content == "!quote" and message.referenced_message:
        # Get the quoted message
        quoted = message.referenced_message
        await message.channel.send(
            f"{quoted.author} said: {quoted.content}"
        )

client.run()
```
