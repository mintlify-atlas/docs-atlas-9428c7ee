## Overview

The `GuildMember` class represents a user within a specific guild context. It combines a `User` object with guild-specific information like nickname, roles, join date, and voice state.

## Class Definition

```python
from fluxer.models import GuildMember

@dataclass(slots=True)
class GuildMember:
    """Represents a member of a guild."""
```

## Usage Example

```python
import fluxer
from fluxer import Embed
from datetime import datetime, timedelta

client = fluxer.Client(token="your_token")

@client.event
async def on_message(message):
    if not message.guild_id:
        return  # Ignore DMs
    
    # Fetch the member object
    guild = await client.fetch_guild(message.guild_id)
    member = await guild.fetch_member(message.author.id)
    
    if message.content == "!memberinfo":
        # Create an embed with member info
        embed = Embed(
            title=f"Member: {member.display_name}",
            color=0x5865F2
        )
        embed.add_field(name="User ID", value=str(member.user.id))
        embed.add_field(name="Nickname", value=member.nick or "None")
        embed.add_field(name="Roles", value=str(len(member.roles)))
        
        if member.joined_at:
            embed.add_field(name="Joined", value=member.joined_at[:10])
        
        if member.communication_disabled_until:
            embed.add_field(
                name="Timed Out Until",
                value=member.communication_disabled_until,
                inline=False
            )
        
        await message.reply(embed=embed)
    
    elif message.content.startswith("!timeout"):
        # Timeout a mentioned user
        if message.mentions:
            target = await guild.fetch_member(message.mentions[0].id)
            
            # Timeout for 10 minutes
            until = (datetime.utcnow() + timedelta(minutes=10)).isoformat()
            await target.timeout(until=until, reason="Moderator action")
            
            await message.reply(f"Timed out {target.mention} for 10 minutes")
    
    elif message.content.startswith("!role add"):
        # Example: !role add @user 123456789
        parts = message.content.split()
        if len(parts) >= 4 and message.mentions:
            target = await guild.fetch_member(message.mentions[0].id)
            role_id = int(parts[3])
            
            await target.add_role(role_id, reason="Moderator action")
            await message.reply(f"Added role to {target.mention}")

client.run()
```
